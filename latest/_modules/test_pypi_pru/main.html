def my_function(x: int, y: int) -> int:
    """Return the product of two integers.

    Parameters
    ----------
    x : int
        First integer.
    y : int
        Second integer.

    Returns
    -------
    int
        The product :math:`x \times y` computed using NumPy.

    Examples
    --------
    >>> my_function(2, 3)
    6
    """

    return int(prod([x, y]))

def annuity_present_value(
    n: int,
    i: float,
    px: float = 1.0,
    m: int = 1
) -> float:
    """Calculate the present value of an annuity with actuarial notation.

    This function demonstrates complex actuarial mathematics notation in
    docstrings using MathJax with custom macros.

    The present value of a life annuity-due of 1 per period for $n$ periods
    is given by:

    .. math::

        \\ax{n} = \\sum_{k=0}^{n-1} \\vx{k} \\cdot \\npx{k}{x}

    where:
    - $\\ax{n}$ is the annuity-due present value (using custom macro)
    - $\\vx{k} = (1+i)^{-k}$ is the discount factor
    - $\\npx{k}{x}$ is the probability that $(x)$ survives $k$ years

    For an $m$-thly annuity, the present value is:

    .. math::

        \\axi{x}{m} = \\frac{1}{m} \\sum_{j=0}^{mn-1} v^{j/m} \\cdot {}_{j/m}p_x

    The relationship between continuous and discrete annuities:

    .. math::

        \\bar{a}_{x:\\overline{n}|} \\approx \\ax{n} - \\frac{m-1}{2m}

    With mortality, the complete life expectancy is:

    .. math::

        \\ex{x} = \\int_0^{\\infty} {}_tp_x \\, dt = \\sum_{k=0}^{\\infty} \\npx{k}{x}

    Parameters
    ----------
    n : int
        Number of payment periods.
    i : float
        Interest rate per period (e.g., 0.05 for 5%).
    px : float, optional
        Survival probability per period (default: 1.0 for certain annuity).
        For life annuities, use ${}_1p_x$ for age $x$.
    m : int, optional
        Number of payments per period (default: 1 for annual).
        Use $m>1$ for $m$-thly annuities.

    Returns
    -------
    float
        Present value $\\ax{n}$ of the annuity-due.

    Notes
    -----
    The formula assumes:

    1. Payments at the **beginning** of each period (annuity-due)
    2. Constant survival probability $\\npx{k}{x} = (p_x)^k$
    3. Effective interest rate $i$ per period

    For accumulated value, use the relationship:

    .. math::

        \\sx{n} = \\ax{n} \\cdot (1+i)^n

    where $\\sx{n}$ is the accumulated value.

    Examples
    --------
    Certain annuity (no mortality):

    >>> annuity_present_value(n=10, i=0.05)
    8.1078...

    Life annuity with 98% annual survival:

    >>> annuity_present_value(n=20, i=0.04, px=0.98)
    12.8453...

    Quarterly annuity (m=4):

    >>> annuity_present_value(n=5, i=0.06, m=4)
    4.4518...

    See Also
    --------
    For insurance values, the relationship is:

    .. math::

        \\Ax{x} + d \\cdot \\ax{x} = 1

    where $\\Ax{x}$ is the insurance present value and $d = i/(1+i)$.

    References
    ----------
    .. [1] Bowers, N. L., et al. (1997). *Actuarial Mathematics* (2nd ed.).
           Society of Actuaries.
    .. [2] Dickson, D. C., et al. (2009). *Actuarial Mathematics for Life
           Contingent Risks*. Cambridge University Press.
    """
    v = 1 / (1 + i)  # Discount factor

    if m == 1:
        # Annual payments
        pv = 0.0
        for k in range(n):
            pv += (v ** k) * (px ** k)
        return pv
    else:
        # m-thly payments
        pv = 0.0
        for j in range(m * n):
            t = j / m
            pv += (v ** t) * (px ** t) / m
        return pv
